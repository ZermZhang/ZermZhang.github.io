常见shell命令

[toc]

0. 常用命令

# 检查空行
cat $file_name | grep '^$'

# 文件按照行进行去重
cat $file_name | sort -n | uniq

# 对文件进行shuffle处理
shuf $file_name -o $target_file_name

# 检查当前目录的占用空间
du -h --max-depth=1

# 重命名文件夹
## 不需要加-r的参数，直接对文件夹使用mv，可以进行文件夹的重命名
mv $origin_folder_name/ ${new_folder_name}

1. 时间相关

  • 各级时间标识
    • 年：%Y
    • 月：%m
    • 日：%d
    • 时：%H
    • 分：%M
    • 秒：%S
# 获取格式化时间（精确到小时）
date +%Y%m%d%H

# 获取毫秒级时间戳
echo $[$(date +%s%N)/1000000]

2. shell命令行中进行数值计算

# 第一种方法
result=`((${head} - ${tail}))`
# 第二种方法
result=`expr ${head} - ${tail}`
